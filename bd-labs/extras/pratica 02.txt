CREATE TABLE PROJETO(
codigo INTEGER PRIMARY KEY,
nome VARCHAR(30));

CREATE TABLE EMPRESA(
codigo INTEGER PRIMARY KEY,
nome VARCHAR(60),
rua VARCHAR(20),
cidade VARCHAR(20),
estado CHAR(2));

CREATE TABLE EMPREGADO(
codigo INTEGER PRIMARY KEY,
codigo_empresa INTEGER,
codigo_gerente INTEGER,
nome VARCHAR(60),
rua VARCHAR(20),
cidade VARCHAR(20),
estado CHAR(2),
sexo CHAR(1),
salario REAL,
FOREIGN KEY (codigo_empresa) REFERENCES EMPRESA(codigo),
FOREIGN KEY (codigo_gerente) REFERENCES EMPREGADO(codigo));

CREATE TABLE TRABALHA_EM(
codigo_empregado INTEGER,
codigo_projeto INTEGER,
horas INTEGER,
FOREIGN KEY (codigo_empregado) REFERENCES EMPREGADO(codigo),
FOREIGN KEY (codigo_projeto) REFERENCES PROJETO(codigo));
